int main() { int a, b, temp; // Input two numbers from the user printf("Enter first number (a): "); scanf("%d", &a); printf("Enter second number (b): "); scanf("%d", &b); // Display values before swapping printf("\nBefore swapping:\n"); printf("a = %d, b = %d\n", a, b); // Swapping using a temporary variable temp = a; a = b; b = temp; // Display values after swapping printf("\nAfter swapping:\n"); printf("a = %d, b = %d\n", a, b); return 0; }
